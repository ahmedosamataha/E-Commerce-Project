CREATE TABLE IF NOT EXISTS "sellers2" ("id"   SERIAL , "name" VARCHAR(255) NOT NULL, "password" VARCHAR(255) NOT NULL, "email" VARCHAR(255) NOT NULL, "createdAt" TIMESTAMP WITH TIME ZONE NOT NULL, "updatedAt" TIMESTAMP WITH TIME ZONE NOT NULL, PRIMARY KEY ("id"));

CREATE TABLE IF NOT EXISTS "consumers2" ("id"   SERIAL , "name" VARCHAR(255) NOT NULL, "shippingAddress" VARCHAR(255) NOT NULL, "password" VARCHAR(255) NOT NULL, "email" VARCHAR(255) NOT NULL, "gender" BOOLEAN, "lastVisited" VARCHAR(255), "cash" INTEGER NOT NULL, "createdAt" TIMESTAMP WITH TIME ZONE NOT NULL, "updatedAt" TIMESTAMP WITH TIME ZONE NOT NULL, PRIMARY KEY ("id"));

CREATE TABLE IF NOT EXISTS "products2" ("id"   SERIAL , "name" VARCHAR(255) NOT NULL, "tag" VARCHAR(255) NOT NULL, "price" DOUBLE PRECISION NOT NULL, "imageUrl" VARCHAR(255) NOT NULL, "quantity" INTEGER NOT NULL, "description" VARCHAR(255) NOT NULL, "createdAt" TIMESTAMP WITH TIME ZONE NOT NULL, "updatedAt" TIMESTAMP WITH TIME ZONE NOT NULL, "sellerId" INTEGER REFERENCES "sellers2" ("id") ON DELETE SET NULL ON UPDATE CASCADE, PRIMARY KEY ("id"));

CREATE TABLE IF NOT EXISTS "carts2" ("id"  SERIAL , "createdAt" TIMESTAMP WITH TIME ZONE NOT NULL, "updatedAt" TIMESTAMP WITH TIME ZONE NOT NULL, "consumerId" INTEGER REFERENCES "consumers2" ("id") ON DELETE SET NULL ON UPDATE CASCADE, PRIMARY KEY ("id"));

CREATE TABLE IF NOT EXISTS "cartItems2" ("id"  SERIAL , "quantity" INTEGER NOT NULL, "createdAt" TIMESTAMP WITH TIME ZONE NOT NULL, "updatedAt" TIMESTAMP WITH TIME ZONE NOT NULL, "productId" INTEGER REFERENCES "products2" ("id") ON DELETE CASCADE ON UPDATE CASCADE, "cartId" INTEGER REFERENCES "carts2" ("id") ON DELETE CASCADE ON UPDATE CASCADE, UNIQUE ("productId", "cartId"), PRIMARY KEY ("id"));

CREATE TABLE IF NOT EXISTS "orders2" ("id"  SERIAL , "createdAt" TIMESTAMP WITH TIME ZONE NOT NULL, "updatedAt" TIMESTAMP WITH TIME ZONE NOT NULL, "consumerId" INTEGER REFERENCES "consumers2" ("id") ON DELETE SET NULL ON UPDATE CASCADE, PRIMARY KEY ("id"));

CREATE TABLE IF NOT EXISTS "orderItems2" ("id"   SERIAL , "quantity" INTEGER NOT NULL, "discount" DOUBLE PRECISION NOT NULL, "createdAt" TIMESTAMP WITH TIME ZONE NOT NULL, "updatedAt" TIMESTAMP WITH TIME ZONE NOT NULL, "productId" INTEGER REFERENCES "products2" ("id") ON DELETE CASCADE ON UPDATE CASCADE, "orderId" INTEGER REFERENCES "orders2" ("id") ON DELETE CASCADE ON UPDATE CASCADE, UNIQUE ("productId", "orderId"), PRIMARY KEY ("id"));

CREATE TABLE IF NOT EXISTS "reviews2" ("id"   SERIAL , "rate" INTEGER, "text" VARCHAR(255), "consumerName" VARCHAR(255), "createdAt" TIMESTAMP WITH TIME ZONE NOT NULL, "updatedAt" TIMESTAMP WITH TIME ZONE NOT NULL, "productId" INTEGER REFERENCES "products2" ("id") ON DELETE SET NULL ON UPDATE CASCADE, PRIMARY KEY ("id"));

CREATE TABLE IF NOT EXISTS "offers2" ("id"  SERIAL , "quantity" INTEGER NOT NULL, "createdAt" TIMESTAMP WITH TIME ZONE NOT NULL, "updatedAt" TIMESTAMP WITH TIME ZONE NOT NULL, "consumerId" INTEGER REFERENCES "consumers2" ("id") ON DELETE SET NULL ON UPDATE CASCADE, PRIMARY KEY ("id"));
